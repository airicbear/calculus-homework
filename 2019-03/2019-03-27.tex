\begin{problem}{26}
Find the area of the region bounded by $y = 3x^2$ and $y = 9x$
\begin{align}
\descmeet
3x^2 &= 9x \\
3x^2 - 9x &= 0 \\
x^2 - 3x &= 0 \\
x(x - 3) &= 0 \\
x = 0 &\And x = 3 \\
\deschigh
3(1)^2 &< 9(1) \\
\descarea
\int_0^3 9x ~ dx - \int_0^3 3x^2 ~ dx &= \frac{9}{2} x^2 ~ \bigg|_0^3 - x^3 ~ \bigg|_0^3 \\
&= \frac{9}{2} (3)^2 - (3)^3 \\
&= \frac{81}{2} - \frac{54}{2} \\
&= 13 \frac{1}{2}
\end{align}
\end{problem}

\end{align}
