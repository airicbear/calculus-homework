\begin{problem}{9}
$y = 4 - x^2; \quad [-2, 2]$
\begin{align}
F(x) &= 4x - \dfrac{x^3}{3} \\
&= \left(4(2) - \dfrac{(2)^3}{3}\right) - \left(4(-2) - \dfrac{(-2)^3}{3}\right) \\
&= 10 \dfrac{2}{3}
\end{align}
\end{problem}
